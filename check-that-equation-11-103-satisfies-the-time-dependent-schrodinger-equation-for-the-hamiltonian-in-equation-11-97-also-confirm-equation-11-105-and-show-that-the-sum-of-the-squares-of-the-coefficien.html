Guest Question.

Check that Equation 11.103 satisfies the time-dependent Schrödinger equation for the Hamiltonian in Equation 11.97. Also confirm Equation 11.105, and show that the sum of the squares of the coefficients is 1, as required for normalization.

\( \chi(t)=\left(\begin{array}{l} {\left[\cos (\lambda t / 2)-i \frac{\left(\omega_{1}-\omega\right)}{\lambda} \sin (\lambda t / 2)\right] \cos (\alpha / 2) e^{-i \omega t / 2}} \\ {\left[\cos (\lambda t / 2)-i \frac{\left(\omega_{1}+\omega\right)}{\lambda} \sin (\lambda t / 2)\right] \sin (\alpha / 2) e^{+i \omega t / 2}} \end{array}\right) \) (11.103).

\( \hat{H}(t)=\frac{e}{m} B \cdot S =\frac{e \hbar B_{0}}{2 m}\left[\sin \alpha \cos (\omega t) \sigma_{x}+\sin \alpha \sin (\omega t) \sigma_{y}+\cos \alpha \sigma_{z}\right] \)

\( =\frac{\hbar \omega_{1}}{2}\left(\begin{array}{cc} \cos \alpha & e^{-i \omega t} \sin \alpha \\ e^{i \omega t} \sin \alpha & -\cos \alpha \end{array}\right) \) (11.97).

\( \chi(t)=\left[\cos \left(\frac{\lambda t}{2}\right)-i \frac{\left(\omega_{1}-\omega \cos \alpha\right)}{\lambda} \sin \left(\frac{\lambda t}{2}\right)\right] e^{-i \omega t / 2} \chi+(t) \)

\( +i\left[\frac{\omega}{\lambda} \sin \alpha \sin \left(\frac{\lambda t}{2}\right)\right] e^{+i \omega t / 2} \chi-(t) \) (11.105).

Admin
Solution

To show: \( i \hbar \frac{\partial \chi}{\partial t}=H \chi \), where χ is given by Eq. 11.103 and H is given by Eq. 11.97:

\( \chi(t)=\left(\begin{array}{l} {\left[\cos (\lambda t / 2)-i \frac{\left(\omega_{1}-\omega\right)}{\lambda} \sin (\lambda t / 2)\right] \cos (\alpha / 2) e^{-i \omega t / 2}} \\ {\left[\cos (\lambda t / 2)-i \frac{\left(\omega_{1}+\omega\right)}{\lambda} \sin (\lambda t / 2)\right] \sin (\alpha / 2) e^{+i \omega t / 2}} \end{array}\right) \) (11.103).

\( \hat{H}(t)=\frac{e}{m} B \cdot S =\frac{e \hbar B_{0}}{2 m}\left[\sin \alpha \cos (\omega t) \sigma_{x}+\sin \alpha \sin (\omega t) \sigma_{y}+\cos \alpha \sigma_{z}\right] \)

\( =\frac{\hbar \omega_{1}}{2}\left(\begin{array}{cc} \cos \alpha & e^{-i \omega t} \sin \alpha \\ e^{i \omega t} \sin \alpha & -\cos \alpha \end{array}\right) \) (11.97).

\(\frac{\partial \chi}{\partial t}= \)

\( \left(\begin{array}{c} \frac{\lambda}{2}\left[-\sin \left(\frac{\lambda t}{2}\right)-i \frac{\left(\omega_{1}-\omega\right)}{\lambda} \cos \left(\frac{\lambda t}{2}\right)\right] \cos \left(\frac{\alpha}{2}\right) e^{-i \omega t / 2}-\frac{i \omega}{2}\left[\cos \left(\frac{\lambda t}{2}\right)-\frac{i\left(\omega_{1}-\omega\right)}{\lambda} \sin \left(\frac{\lambda t}{2}\right)\right] \cos \left(\frac{\alpha}{2}\right) e^{-i \omega t / 2} \\ \frac{\lambda}{2}\left[-\sin \left(\frac{\lambda t}{2}\right)-i \frac{\left(\omega_{1}+\omega\right)}{\lambda} \cos \left(\frac{\lambda t}{2}\right)\right] \sin \left(\frac{\alpha}{2}\right) e^{i \omega t / 2}+\frac{i \omega}{2}\left[\cos \left(\frac{\lambda t}{2}\right)-\frac{i\left(\omega_{1}+\omega\right)}{\lambda} \sin \left(\frac{\lambda t}{2}\right)\right] \sin \left(\frac{\alpha}{2}\right) e^{i \omega t / 2} \end{array}\right) \).

\( H \chi= \)

\( \frac{\hbar \omega_{1}}{2}\left(\begin{array}{l} \cos \alpha\left[\cos \left(\frac{\lambda t}{2}\right)-i \frac{\left(\omega_{1}-\omega\right)}{\lambda} \sin \left(\frac{\lambda t}{2}\right)\right] \cos \frac{\alpha}{2} e^{-i \omega t / 2}+e^{-i \omega t} \sin \alpha\left[\cos \left(\frac{\lambda t}{2}\right)-\frac{i\left(\omega_{1}+\omega\right)}{\lambda} \sin \left(\frac{\lambda t}{2}\right)\right] \sin \frac{\alpha}{2} e^{i \omega t / 2} \\ e^{i \omega t} \sin \alpha\left[\cos \left(\frac{\lambda t}{2}\right)-i \frac{\left(\omega_{1}-\omega\right)}{\lambda} \sin \left(\frac{\lambda t}{2}\right)\right] \cos \frac{\alpha}{2} e^{-i \omega t / 2}-\cos \alpha\left[\cos \left(\frac{\lambda t}{2}\right)-\frac{i\left(\omega_{1}+\omega\right)}{\lambda} \sin \left(\frac{\lambda t}{2}\right)\right] \sin \frac{\alpha}{2} e^{i \omega t / 2} \end{array}\right) \).

(1) Upper elements:

\( i \cancel{\hbar}\left\{\frac{\lambda}{\cancel{2}}\left[-\sin \left(\frac{\lambda t}{2}\right)-i \frac{\left(\omega_{1}-\omega\right)}{\lambda} \cos \left(\frac{\lambda t}{2}\right)\right] \cancel{\cos \frac{\alpha}{2}}-\frac{i \omega}{2}\left[\cos \left(\frac{\lambda t}{2}\right)-\frac{i\left(\omega_{1}-\omega\right)}{\lambda} \sin \left(\frac{\lambda t}{2}\right)\right] \cancel{\cos \frac{\alpha}{2}}\right\} \).

\( \stackrel{?}{=} \frac{\cancel {\hbar} \omega_{1}}{\cancel {2}}\{\left[\cos \left(\frac{\lambda t}{2}\right)-i \frac{\left(\omega_{1}-\omega\right)}{\lambda} \sin \left(\frac{\lambda t}{2}\right)\right] \cos \alpha \cancel {\cos \frac{\alpha}{2}}+\left[\cos \left(\frac{\lambda t}{2}\right)-i \frac{\left(\omega_{1}+\omega\right)}{\lambda} \sin \left(\frac{\lambda t}{2}\right)\right] \underbrace{\sin \alpha}_{\star} \sin \frac{\alpha}{2}\}, \).

\( \text { where } \star =2 \sin \frac{\alpha}{2} \cancel {\cos \frac{ \alpha }{2}}\),

The sine terms:

\( \sin \left(\frac{\lambda t}{2}\right)\left[-i \lambda-\frac{i \omega\left(\omega_{1}-\omega\right)}{\lambda}+\frac{i \omega_{1}\left(\omega_{1}-\omega\right)}{\lambda} \cos \alpha+\frac{i \omega_{1}\left(\omega_{1}+\omega\right)}{\lambda} 2 \sin ^{2} \frac{\alpha}{2}\right] \stackrel{?}{=} 0 \).

\( \frac{i}{\lambda} \sin \left(\frac{\lambda t}{2}\right)\left[\cancel {\omega^{2}}-\omega_{1}^{2}+2 \omega \omega_{1} \cos \alpha-\omega \omega_{1}+\cancel {\omega^{ 2 }}+\left(\omega_{1}^{2}-\omega \omega_{1}\right) \cos \alpha+\left(\omega_{1}^{2}+\omega \omega_{1}\right)(1-\cos \alpha)\right] \).

\( =-\frac{i}{\lambda} \sin \left(\frac{\lambda t}{2}\right)\left[\cancel {\omega_{1}^{2}}+2 \omega \omega_{1} \cos \alpha-\cancel {\omega \omega_{1}}+\cancel {\omega_{1}^{2} \cos \alpha }-\omega \omega_{1} \cos \alpha+\cancel {\omega_{1}^{2}}+\cancel {\omega \omega_{1}}-\cancel {\omega_{1}^{2} \cos \alpha}-\omega \omega_{1} \cos \alpha\right]=0 \)

The cosine terms:

\( \cos \left(\frac{\lambda t}{2}\right)\left[\left(\omega_{1}-\cancel {\omega}\right)+\cancel {\omega}-\omega_{1} \cos \alpha-\omega_{1} 2 \sin ^{2} \frac{\alpha}{2}\right]=-\omega_{1} \cos \left(\frac{\lambda t}{2}\right)[-1+\cos \alpha+(1-\cos \alpha)]=0 \).

(2) Lower elements:

\( i \cancel {\hbar}\left\{\frac{\lambda}{\cancel {2}}\left[-\sin \left(\frac{\lambda t}{2}\right)-i \frac{\left(\omega_{1}+\omega\right)}{\lambda} \cos \left(\frac{\lambda t}{2}\right)\right] \cancel {\sin \left(\frac{\alpha}{2}\right)}+\frac{i \omega}{ 2 }\left[\cos \left(\frac{\lambda t}{2}\right)-\frac{i\left(\omega_{1}+\omega\right)}{\lambda} \sin \left(\frac{\lambda t}{2}\right)\right] \cancel {\sin \left(\frac{\alpha}{2}\right)}\right\} \).

\( \stackrel{?}{=} \frac{\cancel {\hbar }\omega_{1}}{\cancel {2}}\left\{\left[\cos \left(\frac{\lambda t}{2}\right)-i \frac{\left(\omega_{1}-\omega\right)}{\lambda} \sin \left(\frac{\lambda t}{2}\right)\right] 2 \cancel {\sin \frac{\alpha}{2}} \cos ^{2} \frac{\alpha}{2}-\left[\cos \left(\frac{\lambda t}{2}\right)-\frac{i\left(\omega_{1}+\omega\right)}{\lambda} \sin \left(\frac{\lambda t}{2}\right)\right] \cos \alpha \cancel {\sin \frac{\alpha}{2}}\right\} \)

The sine terms:

\( \sin \left(\frac{\lambda t}{2}\right)\left[-i \lambda+\frac{i \omega\left(\omega_{1}+\omega\right)}{\lambda}+\frac{i \omega_{1}\left(\omega_{1}-\omega\right)}{\lambda} 2 \cos ^{2}\left(\frac{\alpha}{2}\right)-\frac{i \omega_{1}\left(\omega_{1}+\omega\right)}{\lambda} \cos \alpha\right] \stackrel{?}{=} 0 \).

\( \frac{i}{\lambda} \sin \left(\frac{\lambda t}{2}\right)\left[-\cancel {\omega^{2}}-\omega_{1}^{2}+2 \omega \omega_{1} \cos \alpha+\omega \omega_{1}+\cancel {\omega^{2}}+\left(\omega_{1}^{2}-\omega \omega_{1}\right)(1+\cos \alpha)-\left(\omega_{1}^{2}+\omega \omega_{1}\right) \cos \alpha\right] \).

\( =\frac{i}{\lambda} \sin \left(\frac{\lambda t}{2}\right)\left[-\cancel {\omega_{1}^{2}}+2 \omega \omega_{1} \cos \alpha+\cancel {\omega \omega_{1}}+\cancel {\omega_{1}^{2}}-\cancel {\omega \omega_{1}}+\cancel {\omega_{1}^{2} \cos \alpha}-\omega \omega_{1} \cos \alpha-\cancel {\omega_{1}^{2} \cos \alpha}-\omega \omega_{1} \cos \alpha\right]=0 \).

The cosine terms:

\( \cos \left(\frac{\lambda t}{2}\right)\left[\left(\omega_{1}+\cancel {\omega}\right)-\cancel {\omega}-\omega_{1} 2 \cos ^{2} \frac{\alpha}{2}+\omega_{1} \cos \alpha\right]=\cos \left(\frac{\lambda t}{2}\right)\left[\omega_{1}-\omega_{1}(1+\cos \alpha)+\omega_{1} \cos \alpha\right]=0 \).

As for Eq. 11.105:

\( \chi(t)=\left[\cos \left(\frac{\lambda t}{2}\right)-i \frac{\left(\omega_{1}-\omega \cos \alpha\right)}{\lambda} \sin \left(\frac{\lambda t}{2}\right)\right] e^{-i \omega t / 2} \chi+(t) \)

\( +i\left[\frac{\omega}{\lambda} \sin \alpha \sin \left(\frac{\lambda t}{2}\right)\right] e^{+i \omega t / 2} \chi-(t) \) (11.105).

\( \left[\cos \left(\frac{\lambda t}{2}\right)-i \frac{\left(\omega_{1}-\omega \cos \alpha\right)}{\lambda} \sin \left(\frac{\lambda t}{2}\right)\right] e^{-i \omega t / 2}\left(\begin{array}{c} \cos \frac{\alpha}{2} \\ e^{i \omega t} \sin \frac{\alpha}{2}\end{array}\right)+i\left[\frac{\omega}{\lambda} \sin \alpha \sin \left(\frac{\lambda t}{2}\right)\right] e^{-i \omega t / 2}\left(\begin{array}{c} \sin \frac{\alpha}{2} \\ -e^{i \omega t} \cos \frac{\alpha}{2} \end{array}\right) \).

\( =\left(\begin{array}{l} \alpha \\ \beta \end{array}\right) \text {, with } \)

\( \alpha=\left\{\left[\cos \left(\frac{\lambda t}{2}\right)-\frac{i \omega_{1}}{\lambda} \sin \left(\frac{\lambda t}{2}\right)\right] \cos \frac{\alpha}{2}+\frac{i \omega}{\lambda}[\underbrace{\cos \alpha \cos \frac{\alpha}{2}+\sin \alpha \sin \frac{\alpha}{2}}_{\cos \left(\alpha-\frac{\alpha}{2}\right)=\cos \frac{\alpha}{2}}] \sin \left(\frac{\lambda t}{2}\right)\right\} e^{-i \omega t / 2} \).

\( =\left[\cos \left(\frac{\lambda t}{2}\right)-\frac{i\left(\omega_{1}-\omega\right)}{\lambda} \sin \left(\frac{\lambda t}{2}\right)\right] \cos \frac{\alpha}{2} e^{-i \omega t / 2} \) (confirming the top entry).

\( \beta=\left\{\left[\cos \left(\frac{\lambda t}{2}\right)-\frac{i \omega_{1}}{\lambda} \sin \left(\frac{\lambda t}{2}\right)\right] \sin \frac{\alpha}{2}+\frac{i \omega}{\lambda}[\underbrace{\cos \alpha \sin \frac{\alpha}{2}-\sin \alpha \cos \frac{\alpha}{2}}_{\sin \left(\frac{\alpha}{2}-\alpha\right)=-\sin \frac{\alpha}{2}}] \sin \left(\frac{\lambda t}{2}\right)\right\} e^{i \omega t / 2} \).

\( =\left[\cos \left(\frac{\lambda t}{2}\right)-\frac{i\left(\omega_{1}+\omega\right)}{\lambda} \sin \left(\frac{\lambda t}{2}\right)\right] \sin \frac{\alpha}{2} e^{i \omega t / 2} \). (confirming the bottom entry).

\( \left|c_{+}\right|^{2}+\left|c_{-}\right|^{2}=\cos ^{2}\left(\frac{\lambda t}{2}\right)+\frac{\left(\omega_{1}-\omega \cos \alpha\right)^{2}}{\lambda^{2}} \sin ^{2}\left(\frac{\lambda t}{2}\right)+\frac{\omega^{2}}{\lambda^{2}} \sin ^{2} \alpha \sin ^{2}\left(\frac{\lambda t}{2}\right) \).

\( =\cos ^{2}\left(\frac{\lambda t}{2}\right)+\frac{1}{\lambda^{2}}(\underbrace{\omega_{1}^{2}-2 \omega \omega_{1} \cos \alpha+\omega^{2} \cos ^{2} \alpha+\omega^{2} \sin ^{2} \alpha}_{\omega^{2}+\omega_{1}^{2}-2 \omega \omega_{1} \cos \alpha=\lambda^{2}}) \sin ^{2}\left(\frac{\lambda t}{2}\right) \).

\( =\cos ^{2}\left(\frac{\lambda t}{2}\right)+\sin ^{2}\left(\frac{\lambda t}{2}\right)=1 \).
